[PLAINTEXT] {
//Insertion sort
Iterate over the array starting with the second element
	use this element as the insertion value
	iterate over previous elements of the array reverse
		if the element to the left is greater
			set the value of the current element to the value of the left element
		else
			insert value at current element
			break out of loop
		if current index is 0
			insert value
			break out of loop
			
//Selection Sort
iterate over every element
	select value at that element
	for each element to the right
		compare selected to with value at that element
			if selected value is greater
				swap selected value with value at current index
			else continue
	insert selection into current selection index
}
